export function DownloadButton({ files }) {
  const [open, setOpen] = React.useState(false);
  const ref = React.useRef(null);
  React.useEffect(() => {
    if (!open) return;
    const handler = (e) => {
      if (ref.current && !ref.current.contains(e.target)) setOpen(false);
    };
    document.addEventListener('mousedown', handler);
    return () => document.removeEventListener('mousedown', handler);
  }, [open]);
  return (
    <div ref={ref} style={{ position: 'relative' }}>
      <button
        onClick={() => setOpen(!open)}
        style={{
          background: 'none',
          border: '1px solid var(--mieweb-border, #d1d5db)',
          borderRadius: '6px',
          padding: '4px 8px',
          cursor: 'pointer',
          fontSize: '13px',
          display: 'flex',
          alignItems: 'center',
          gap: '4px',
        }}
      >
        <svg
          width="14"
          height="14"
          viewBox="0 0 24 24"
          fill="none"
          stroke="currentColor"
          strokeWidth="2"
          strokeLinecap="round"
          strokeLinejoin="round"
        >
          <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4" />
          <polyline points="7 10 12 15 17 10" />
          <line x1="12" y1="15" x2="12" y2="3" />
        </svg>
        Download
      </button>
      {open && (
        <div
          style={{
            position: 'absolute',
            right: 0,
            top: '100%',
            marginTop: '4px',
            background: 'var(--mieweb-background, #fff)',
            border: '1px solid var(--mieweb-border, #d1d5db)',
            borderRadius: '8px',
            boxShadow: '0 4px 12px rgba(0,0,0,0.1)',
            zIndex: 10,
            minWidth: '160px',
            overflow: 'hidden',
          }}
        >
          {files.map((f, i) => (
            <a
              key={f.path}
              href={f.path}
              download
              style={{
                display: 'block',
                padding: '8px 12px',
                fontSize: '13px',
                textDecoration: 'none',
                color: 'inherit',
                borderTop:
                  i > 0 ? '1px solid var(--mieweb-border, #e5e7eb)' : 'none',
              }}
              onMouseEnter={(e) =>
                (e.target.style.backgroundColor =
                  'var(--mieweb-accent, #f3f4f6)')
              }
              onMouseLeave={(e) =>
                (e.target.style.backgroundColor = 'transparent')
              }
            >
              {f.label}
            </a>
          ))}
        </div>
      )}
    </div>
  );
}

export function LogoCard({ name, brand, lm, dm, color }) {
  const lightSrc = lm ? `/${brand}/${lm}` : null;
  const darkSrc = dm ? `/${brand}/${dm}` : null;
  const files = [
    lm && { path: `/${brand}/${lm}`, label: 'SVG — Light' },
    dm && { path: `/${brand}/${dm}`, label: 'SVG — Dark' },
  ].filter(Boolean);
  return (
    <div
      style={{
        borderRadius: '12px',
        border: '1px solid var(--mieweb-border, #e5e7eb)',
      }}
    >
      <div
        style={{
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          padding: '24px',
          backgroundColor: '#ffffff',
          minHeight: '120px',
          borderRadius: '12px 12px 0 0',
        }}
      >
        {lightSrc ? (
          <img
            src={lightSrc}
            alt={`${name} logo (light)`}
            style={{ height: '64px', maxWidth: '100%', objectFit: 'contain' }}
          />
        ) : (
          <span
            style={{ fontSize: '13px', color: '#9ca3af', fontStyle: 'italic' }}
          >
            Light variant pending
          </span>
        )}
      </div>
      <div
        style={{
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'center',
          padding: '24px',
          backgroundColor: '#1a1a2e',
          minHeight: '120px',
        }}
      >
        {darkSrc ? (
          <img
            src={darkSrc}
            alt={`${name} logo (dark)`}
            style={{ height: '64px', maxWidth: '100%', objectFit: 'contain' }}
          />
        ) : (
          <span
            style={{ fontSize: '13px', color: '#6b7280', fontStyle: 'italic' }}
          >
            Dark variant pending
          </span>
        )}
      </div>
      <div
        style={{
          padding: '12px 16px',
          display: 'flex',
          alignItems: 'center',
          justifyContent: 'space-between',
          borderTop: '1px solid var(--mieweb-border, #e5e7eb)',
        }}
      >
        <div>
          <strong style={{ display: 'block', fontSize: '14px' }}>{name}</strong>
          <span
            style={{ fontSize: '12px', fontFamily: 'monospace', color: color }}
          >
            {brand}/
          </span>
        </div>
        <DownloadButton files={files} />
      </div>
    </div>
  );
}
